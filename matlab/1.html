function [] = square_volume()
t = 0:0.1:2*pi;
z = t - sin(t);
alpha = -pi:0.1:pi+0.1;
x = (cos(alpha)')*z;
y = (sin(alpha)')*z;
t = kron(ones(length(alpha), 1), t);
 
mesh(x,y,t)

hold on
axis equal

t = 0:0.1:2*pi;
z = 1 - cos(t);
alpha = -pi:0.1:pi+0.1;
x = (cos(alpha)')*z;
y = (sin(alpha)')*z;
t = kron(ones(length(alpha), 1), t);
 
mesh(x,y,t)

q = 2.*pi.*quad('y = (x-sin(x)).*sqrt((1-cos(x)).^2 + (sin(x)).^2)', 0, 2*pi);
disp(q);
draw
endfunction
